#!/bin/bash


MENU=(
    "Проверка email"
    "Проверка url"
    "Проверка numb"
    "Проверка pass"
    "Завершение работы"
)
select menu in "${MENU[@]}" ; do
    case $REPLY in
        1) . ./email.sh ;;
        2) . ./url.sh ;;
        3) . ./numb ;;
        4) . ./pass ;;
        5) exit ;;
        6) break ;;
        *)
        echo "Недопустимая опция ${MENU[@]}"
    esac
done
