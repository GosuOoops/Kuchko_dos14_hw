#!/bin/bash

email=$1

if [[ "$email" =~ ^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,4}$ ]]

#^ : означает начать с

#[A-Za-z0-9._%+-] : означает любую заглавную, маленькую, цифры, специальные символы.

#+ : означает один или несколько предыдущих.

#@ : означает, что он должен иметь символ @

#[A-Za-z0-9.-]+ : означает буквы и цифры, а точки или тире должны
#быть найдены одним или несколькими (что + означает ).

#\. : означает должен использовать точку, используя обратную косую
#черту спереди, чтобы использовать ее как обычную точку, а не как
#регулярное выражение.

#[A-Za-z]{2,4}$ : означает, что он должен заканчиваться словом
#буксировочных букв только до четырех букв.

then
    echo "Email address $email is valid."
else
    echo "Email address $email is invalid."
fi
